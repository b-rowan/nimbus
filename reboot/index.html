    Reboot¤

    Rebooting and restarting are slightly different. The reboot command is a full device reboot. The restart command will stop the mining process then start it again.

    See the restart page for more information.

    Example¤

    Request¤

    CGMiner style command

    Reboot now: 

    {'command': 'reboot'}

    Example 
    echo '{"command":"reboot"}' | nc {DEVICE_IP} 4028

    import socket
import json


def main():
    with socket.create_connection(({DEVICE_IP}, 4028)) as sock:
        cmd_str = json.dumps({"command":"reboot"})
        sock.sendall(cmd_str.encode("utf-8"))

        response = b""
        while True:
            chunk = sock.recv(4096)
            if not chunk:
                break
            response += chunk

    return response.decode("utf-8")


if __name__ == "__main__":
    response = main()
    print(response)

    Wait 10 seconds before rebooting: 

    {'command': 'reboot', 'param': {'after': 10}}

    Example 
    echo '{"command":"reboot","param":{"after":10}}' | nc {DEVICE_IP} 4028

    POST request

    Reboot now: 

    /nimbus/v0/reboot

    Example 
    curl \
    --request POST \
    --data {"after": 0} \
    http://{DEVICE_IP}/nimbus/v0-1-0/reboot

    import httpx
import asyncio

URL = f"http://{DEVICE_IP}/nimbus/v0-1-0/reboot"

async def main():
    async with httpx.AsyncClient() as c:
        result = await c.post(URL, data={'after': 0})
    json_data = result.json()
    print(json_data)
    return json_data

if __name__ == '__main__':
    asyncio.run(main())

    class NimbusRebootResult(BaseModel):
    """
    The result of the reboot command.
    """

    model_config = ConfigDict(populate_by_name=True, alias_generator=to_cgminer)

    when: Annotated[datetime, BeforeValidator(parse_unix_timestamp)]
    """
    When the reboot will occur as UNIX timestamp in seconds.
    """

    @field_serializer("when")
    def serialize_when(self, when: datetime, _info) -> int:
        return int(when.timestamp())
